 6int model_function(char* in_buf, uint64_t length_in_buf,
 7		   char** out_buf, uint64_t* length_out_buf) {
 8  ygg_init();
 9  int error_code = 0;
10  int nthreads = atoi(getenv("NTHREAD"));
11#ifdef _OPENMP
12  omp_set_num_threads(nthreads);
13#pragma omp parallel for shared(error_code)
14#endif
15  for (int i = 0; i < nthreads; ++i){
16    int flag;
17#ifdef _OPENMP
18#pragma omp critical
19    {
20#endif
21      flag = error_code;
22#ifdef _OPENMP
23    }
24#endif
25    if (flag == 0) {
26      char* out_temp = NULL;
27      uint64_t length_out_temp = 0;
28      char** out_buf_temp = &out_temp;
29      uint64_t* length_out_buf_temp = &length_out_temp;
30      if (i == 0) {
31	out_buf_temp = out_buf;
32	length_out_buf_temp = length_out_buf;
33      }
34      
35      // The WITH_GLOBAL_SCOPE macro is required to ensure that the
36      // comm persists between function calls
37      WITH_GLOBAL_SCOPE(yggRpc_t rpc = yggRpcClient("server_client", "%s", "%s"));
38      printf("client(C:%d): Sending %s (length = %d)\n",
39	     i, in_buf, (int)(length_in_buf));
40      int ret = rpcCallRealloc(rpc, in_buf, length_in_buf,
41			       out_buf_temp, length_out_buf_temp);
42      printf("client(C:%d): Received %s (length = %d)\n",
43	     i, in_buf, (int)(length_in_buf));
44      if (ret < 0) {
45	printf("client(C:%d): RPC CALL ERROR\n", i);
46#ifdef _OPENMP
47#pragma omp critical
48	{
49#endif
50	  error_code = -1;
51#ifdef _OPENMP
52	}
53#endif
54      }
55      if (i != 0) {
56	free(out_temp);
57      }
58    }
59  }
60  return error_code;
61}
