model_error¶

A single model that raises/returns an error.

int main(int argc, char *argv[]) {
  // Return a non-zero value to indicate an error

model:
  name: model_error_c
  driver: GCCModelDriver
  args: ./src/model_error.c  # compile the C code with necessary libraries

  inputs:
    - name: inFile
      driver: FileInputDriver
      args: ./Input/input.txt
    - name: model_errorQueueIn
      driver: InputDriver
      args: model_errorQueue
  
  outputs:
    - name: model_errorQueueOut
      driver: OutputDriver
      args: model_errorQueue
    - name: outFile
      driver: FileOutputDriver
      args: output_model_error.txt
      in_temp: True

  SetErrorMode(SEM_FAILCRITICALERRORS | SEM_NOGPFAULTERRORBOX);
  _set_abort_behavior(0,_WRITE_ABORT_MSG);
#endif
  // Throw an error or return a non-zero value to indicate an error
  throw "Test error";
  return -1;
}
