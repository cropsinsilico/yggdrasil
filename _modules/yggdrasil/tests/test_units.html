        r"""Test convert_R_unit_string."""
        pairs = [('g', 'g'), ('g2', '(g**2)'),
                 ('g2 km s-2', '(g**2)*km*(s**-2)'),
                 ('degC d', 'degC*d'),
                 (tools.bytes2str(b'\xc2\xb0C d'),
                  tools.bytes2str(b'\xc2\xb0C*d')),
                 ('h', 'hr'),
                 ('hrs/kg', 'hr/kg'),
                 ('', ''),
                 ('cm**(-2)', '(cm**-2)')]
        for x, y in pairs:
            self.assert_equal(units.convert_R_unit_string(x), y)
            self.assert_equal(units.convert_R_unit_string(y), y)
            units.add_units(1.0, x)
